## 📝 Dev Notes

**Example: Modal.test.tsx**
```typescript
import { render, screen, fireEvent } from '@testing-library/react';
import userEvent from '@testing-library/user-event';
import { Modal } from '@/components/Modal';

describe('Modal', () => {
  it('renders modal with title and content', () => {
    render(
      <Modal title="Test Modal" isOpen={true} onClose={() => {}}>
        <p>Modal content</p>
      </Modal>
    );
    expect(screen.getByText('Test Modal')).toBeInTheDocument();
    expect(screen.getByText('Modal content')).toBeInTheDocument();
  });

  it('closes modal when close button is clicked', async () => {
    const onClose = vi.fn();
    render(
      <Modal title="Test" isOpen={true} onClose={onClose}>
        Content
      </Modal>
    );
    const closeButton = screen.getByRole('button', { name: /close/i });
    await userEvent.click(closeButton);
    expect(onClose).toHaveBeenCalled();
  });

  it('has focus trap enabled', () => {
    const { container } = render(
      <Modal title="Test" isOpen={true} onClose={() => {}}>
        Content
      </Modal>
    );
    const modalDiv = container.querySelector('[role="dialog"]');
    expect(modalDiv).toHaveAttribute('aria-modal', 'true');
  });

  it('closes on ESC key', async () => {
    const onClose = vi.fn();
    render(
      <Modal title="Test" isOpen={true} onClose={onClose}>
        Content
      </Modal>
    );
    fireEvent.keyDown(document, { key: 'Escape' });
    expect(onClose).toHaveBeenCalled();
  });
});
```

**Example: BrutalButton.test.tsx**
```typescript
import { render, screen } from '@testing-library/react';
import userEvent from '@testing-library/user-event';
import { BrutalButton } from '@/components/BrutalButton';

describe('BrutalButton', () => {
  it('renders button with text', () => {
    render(<BrutalButton>Click me</BrutalButton>);
    expect(screen.getByRole('button', { name: 'Click me' })).toBeInTheDocument();
  });

  it('calls onClick handler when clicked', async () => {
    const onClick = vi.fn();
    render(<BrutalButton onClick={onClick}>Click</BrutalButton>);
    await userEvent.click(screen.getByRole('button'));
    expect(onClick).toHaveBeenCalled();
  });

  it('respects disabled state', () => {
    const { container } = render(<BrutalButton disabled>Click</BrutalButton>);
    const button = screen.getByRole('button');
    expect(button).toBeDisabled();
  });

  it('has accessible aria-label', () => {
    render(<BrutalButton aria-label="Settings">⚙️</BrutalButton>);
    expect(screen.getByLabelText('Settings')).toBeInTheDocument();
  });
});
```
